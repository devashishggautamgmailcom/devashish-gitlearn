-----------------
Git Commands
-----------------
$ git --version
$ git config --global user.name "Devashish"
$ git config --global user.email "devashishgautam8@gmail.com"
$ git config --list
$ git init
$ git status
$ git add  first.html
$ git commit -m "First Commit On first.html file"
$ git add -A
$ git commit -m "2nd Commit On All file" 
$ git checkout first.html
$ git diff
$ git diff --staged
$ git diff --HEAD // STAGING Area changes
$ git diff HEAD // Commit Area changes
$ git log (ALL Log Details)
$ git log  -p -2 (Last 2 Log Details)
$ git log --stat
$ git rm --cached extra.html
$ git branch --list
$ git rm -f extra.html
$ git branch --list
$ git branch user1_branch1
$ git checkout user1_branch1
$ git merge user1_branch1

$ git remote add origin https://github.com/devashishggautamgmailcom/devashish-gitlearn.git

$ git remote -v
origin  https://github.com/devashishggautamgmailcom/devashish-gitlearn.git (fetch)
origin  https://github.com/devashishggautamgmailcom/devashish-gitlearn.git (push)

$ git branch -M main
$ git push -u origin main
